        preferredTimeSlot:
          type: string

        travelRequirements:
          type: string
          enum: [LOCAL, ACCOMMODATION, VIRTUAL]
        technicalRequirements:
          type: array
          items:
            type: string
            enum: [MAC_ADAPTER, REMOTE_OPTION, SPECIAL_AV]

          type: string
          maxLength: 200

          type: string
