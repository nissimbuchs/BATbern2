### Database Migration Update

**File**: `services/event-management-service/src/main/resources/db/migration/VX__update_events_meaningful_id.sql`

```sql
-- Add eventCode and eventNumber columns
ALTER TABLE events
  ADD COLUMN event_code VARCHAR(50),
  ADD COLUMN event_number INTEGER;

-- Populate eventNumber from existing data (sequential)
UPDATE events
SET event_number = ROW_NUMBER() OVER (ORDER BY created_at);

-- Populate eventCode from eventNumber
UPDATE events
SET event_code = 'BATbern' || event_number;

-- Add constraints
ALTER TABLE events
  ALTER COLUMN event_code SET NOT NULL,
  ALTER COLUMN event_number SET NOT NULL,
  ADD CONSTRAINT uq_event_code UNIQUE (event_code),
  ADD CONSTRAINT uq_event_number UNIQUE (event_number);

-- Create indexes
CREATE UNIQUE INDEX idx_event_code ON events(event_code);
CREATE UNIQUE INDEX idx_event_number ON events(event_number);

-- Create trigger for auto-generation
CREATE OR REPLACE FUNCTION generate_event_code()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.event_code IS NULL AND NEW.event_number IS NOT NULL THEN
        NEW.event_code := 'BATbern' || NEW.event_number;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_generate_event_code
    BEFORE INSERT ON events
    FOR EACH ROW
    EXECUTE FUNCTION generate_event_code();
```
