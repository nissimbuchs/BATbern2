### 6. Code Patterns

**Speaker Workflow Service Pattern (from 06a-workflow-state-machines.md, lines 99-171):**

```java
// SpeakerWorkflowService.java

@Service
@Slf4j
public class SpeakerWorkflowService {

    private final SessionRepository sessionRepository;
    private final SpeakerRepository speakerRepository;
    private final WorkflowNotificationService notificationService;
    private final DomainEventPublisher eventPublisher;
    private final EventRepository eventRepository;

    public void updateSpeakerWorkflowState(String sessionSlug, String username,
                                         SpeakerWorkflowState newState, String updatedBy) {
        // Note: Uses meaningful IDs per ADR-003
        Session session = sessionRepository.findBySlug(sessionSlug)
            .orElseThrow(() -> new EntityNotFoundException("Session not found: " + sessionSlug));

        SessionSpeaker speaker = session.getSpeakers().stream()
            .filter(s -> s.getUsername().equals(username))
            .findFirst()
            .orElseThrow(() -> new EntityNotFoundException("Speaker not found in session: " + username));

        SpeakerWorkflowState previousState = speaker.getWorkflowState();

        // Apply state-specific logic
        switch (newState) {
            case CONTACTED:
                speaker.setContactedAt(Instant.now());
                notificationService.sendSpeakerInvitation(speaker, session);
                break;
            case ACCEPTED:
                speaker.setConfirmedAt(Instant.now());
                checkForOverflow(session.getEventCode());  // Uses meaningful ID per ADR-003
                break;
            case DECLINED:
                speaker.setDeclinedAt(Instant.now());
                handleSpeakerDecline(session, speaker);
                break;
            case SLOT_ASSIGNED:
                validateSlotAssignment(speaker);
                break;
        }

        speaker.setWorkflowState(newState);
        sessionRepository.save(session);

        // Publish workflow state change event (uses meaningful IDs per ADR-003)
        eventPublisher.publishEvent(new SpeakerWorkflowStateChangeEvent(
            username, sessionSlug, session.getEventCode(), previousState, newState, updatedBy
        ));

        log.info("Speaker {} in session {} moved from {} to {}",
                 username, sessionSlug, previousState, newState);
    }

    private void checkForOverflow(String eventCode) {
        // Uses meaningful ID per ADR-003
        Event event = eventRepository.findByEventCode(eventCode)
            .orElseThrow(() -> new EntityNotFoundException("Event not found: " + eventCode));

        int maxSlots = event.getSlotConfiguration().getMaxSlots();
        long acceptedSpeakers = event.getSessions().stream()
            .flatMap(session -> session.getSpeakers().stream())
            .filter(speaker -> speaker.getWorkflowState() == SpeakerWorkflowState.ACCEPTED)
            .count();

        if (acceptedSpeakers > maxSlots) {
            eventPublisher.publishEvent(new SpeakerOverflowDetectedEvent(
                eventCode, acceptedSpeakers, maxSlots  // Uses meaningful ID per ADR-003
            ));
        }
    }

    private void handleSpeakerDecline(Session session, SessionSpeaker speaker) {
        // Handle decline logic
        log.info("Speaker {} declined invitation for session {}", speaker.getUsername(), session.getSlug());
    }

    private void validateSlotAssignment(SessionSpeaker speaker) {
        // Validate that slot assignment exists
        if (speaker.getSlotAssignment() == null) {
            throw new WorkflowValidationException("Cannot assign SLOT_ASSIGNED state without slot", Map.of());
        }
    }
}
```
