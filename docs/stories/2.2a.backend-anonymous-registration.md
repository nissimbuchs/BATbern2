**Updated UserService.java**:
```java
@Service
public class UserService {
    private final UserRepository userRepository;
    private final CognitoService cognitoService;

    public UserProfile getOrCreateUser(GetOrCreateUserRequest request) {
        boolean shouldSyncCognito = request.cognitoSync() == null || request.cognitoSync();

        // Check if user exists by email
        Optional<UserProfile> existingUser = userRepository.findByEmail(request.email());
        if (existingUser.isPresent()) {
            return existingUser.get();
        }

        UserProfile userProfile = new UserProfile();
        userProfile.setEmail(request.email());
        userProfile.setFirstName(request.firstName());
        userProfile.setLastName(request.lastName());
        userProfile.setCompany(request.company());
        userProfile.setRole(request.role());

        if (shouldSyncCognito) {
            // Existing behavior: fetch from Cognito
            String cognitoId = cognitoService.getCognitoIdByEmail(request.email());
            userProfile.setCognitoUserId(cognitoId);
            userProfile.setUsername(generateUsernameFromCognito(cognitoId));
        } else {
            // ADR-005: Anonymous user
            userProfile.setCognitoUserId(null);
            userProfile.setUsername(generateUsername(request.firstName(), request.lastName(), request.email()));
        }

        return userRepository.save(userProfile);
    }

    private String generateUsername(String firstName, String lastName, String email) {
        String baseUsername = (firstName + "." + lastName)
            .toLowerCase()
            .replaceAll("[^a-z0-9.]", "");

        String username = baseUsername;
        int attempt = 0;

        while (userRepository.existsByUsername(username)) {
            username = baseUsername + "." + RandomStringUtils.randomAlphanumeric(6).toLowerCase();
            attempt++;
            if (attempt > 10) {
                // Fallback to email-based username
                username = email.split("@")[0] + "." + RandomStringUtils.randomAlphanumeric(6).toLowerCase();
                break;
            }
        }

        return username;
    }
}
```

```typescript
// infrastructure/lambda/cognito-post-auth-trigger/index.ts
import { PostAuthenticationTriggerHandler } from 'aws-lambda';
import axios from 'axios';

export const handler: PostAuthenticationTriggerHandler = async (event) => {
  const email = event.request.userAttributes.email;
  const cognitoId = event.request.userAttributes.sub;

  try {
    // Call User Management API to link account
    await axios.patch(
      `${process.env.USER_MANAGEMENT_API_URL}/users/link-cognito`,
      {
        email,
        cognitoUserId: cognitoId,
      },
      {
        headers: {
          'Content-Type': 'application/json',
          // Use internal service authentication
          'X-Service-Token': process.env.SERVICE_TOKEN,
        },
      }
    );

    console.log(`Linked Cognito account for email: ${email}`);
  } catch (error) {
    console.error('Failed to link Cognito account:', error);
    // Don't fail authentication, just log the error
  }

  return event;
};
```

```java
            return user;
```

**RegistrationService.java**:
```java
@Service
public class RegistrationService {
    private final RegistrationRepository registrationRepository;
    private final EventRepository eventRepository;
    private final UserApiClient userApiClient;  // Use existing client
    private final RegistrationEmailService emailService;
    private final QRCodeService qrCodeService;

    public RegistrationDTO createRegistration(String eventCode, CreateRegistrationRequest request) {
        // 1. Find event
        Event event = eventRepository.findByEventCode(eventCode)
            .orElseThrow(() -> new EventNotFoundException(eventCode));

        // 2. Check capacity
        if (event.getCurrentAttendees() >= event.getMaxAttendees()) {
            throw new EventFullException(eventCode);
        }

        // 3. Get or create user profile (anonymous) using existing UserApiClient
        GetOrCreateUserRequest userRequest = GetOrCreateUserRequest.builder()
            .email(request.getEmail())
            .firstName(request.getFirstName())
            .lastName(request.getLastName())
            .company(request.getCompany())
            .role(request.getRole())
            .cognitoSync(false)  // ADR-005: Create anonymous user
            .build();

        UserProfileDTO userProfile = userApiClient.getOrCreateUser(userRequest);

        // 4. Check for duplicate registration
        if (registrationRepository.existsByEventAndAttendeeUsername(event, userProfile.getUsername())) {
            throw new DuplicateRegistrationException(eventCode, request.getEmail());
        }

        // 5. Create registration
        Registration registration = new Registration();
        registration.setEvent(event);
        registration.setAttendeeUsername(userProfile.getUsername());
        registration.setRegistrationCode(generateRegistrationCode(eventCode));
        registration.setStatus(RegistrationStatus.CONFIRMED);
        registration.setRegistrationDate(ZonedDateTime.now());

        Registration saved = registrationRepository.save(registration);

        // 6. Update event attendee count
        event.setCurrentAttendees(event.getCurrentAttendees() + 1);
        eventRepository.save(event);

        // 7. Send confirmation email (async)
        emailService.sendRegistrationConfirmation(saved, userProfile, event);

        // 8. Return enriched DTO
        return enrichRegistration(saved);
    }

    private String generateRegistrationCode(String eventCode) {
        return eventCode + "-reg-" + RandomStringUtils.randomAlphanumeric(6).toLowerCase();
    }

    public RegistrationDTO getRegistrationByCode(String eventCode, String registrationCode) {
        Registration registration = registrationRepository
            .findByRegistrationCode(registrationCode)
            .orElseThrow(() -> new RegistrationNotFoundException(registrationCode));

        // Verify event code matches
        if (!registration.getEvent().getEventCode().equals(eventCode)) {
            throw new RegistrationNotFoundException(registrationCode);
        }

        return enrichRegistration(registration);
    }

    private RegistrationDTO enrichRegistration(Registration registration) {
        // Fetch user details using existing UserApiClient (already cached)
        UserProfileDTO userProfile = userApiClient.getUserByUsername(
            registration.getAttendeeUsername()
        );

        return RegistrationDTO.builder()
            .registrationCode(registration.getRegistrationCode())
            .eventCode(registration.getEvent().getEventCode())
            .attendeeUsername(registration.getAttendeeUsername())
            .attendeeName(userProfile.getFirstName() + " " + userProfile.getLastName())
            .attendeeEmail(userProfile.getEmail())
            .company(userProfile.getCompany())
            .status(registration.getStatus())
            .registrationDate(registration.getRegistrationDate())
            .event(EventSummaryDTO.from(registration.getEvent()))
            .build();
    }
}
```

```java
@Service
public class QRCodeService {

    public byte[] generateQRCode(String data, int width, int height) throws IOException {
        try {
            QRCodeWriter qrCodeWriter = new QRCodeWriter();
            BitMatrix bitMatrix = qrCodeWriter.encode(
                data,
                BarcodeFormat.QR_CODE,
                width,
                height
            );

            ByteArrayOutputStream pngOutputStream = new ByteArrayOutputStream();
            MatrixToImageWriter.writeToStream(bitMatrix, "PNG", pngOutputStream);
            return pngOutputStream.toByteArray();
        } catch (WriterException e) {
            throw new IOException("Failed to generate QR code", e);
        }
    }
```
