```typescript
import { Stepper, Step, StepLabel } from '@mui/material';

const RegistrationWizard = () => {
  const [activeStep, setActiveStep] = useState(0);
  const steps = ['Personal Information', 'Review & Confirm'];

  return (
    <Box>
      <Stepper activeStep={activeStep}>
        {steps.map((label) => (
          <Step key={label}>
            <StepLabel>{label}</StepLabel>
          </Step>
        ))}
      </Stepper>

      {activeStep === 0 && <RegistrationStep1 onContinue={() => setActiveStep(1)} />}
      {activeStep === 1 && <RegistrationStep2 onBack={() => setActiveStep(0)} />}
    </Box>
  );
};
```
