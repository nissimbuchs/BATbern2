## Story

**As an** organizer,
**I want** to track my speaker outreach activities,
**so that** I can coordinate with other organizers and ensure all speakers are contacted.

## Dev Notes - Implementation Guide

### Technical Design Notes

**Outreach Tracking Pattern (AC1-7):**
- **Dashboard Display**: Table with columns: Speaker Name, Company, Assigned Organizer, Last Contact Date, Days Since Assignment, Status
- **Filter by Organizer**: Dropdown with all organizers, filter speaker pool by `assigned_organizer_username`
- **Days Since Assignment**: Calculate `DATEDIFF(CURRENT_DATE, speaker_pool.created_at)`, highlight if > 7 days
- **Bulk Actions**: Checkbox selection with "Mark All Contacted" button, creates multiple outreach records in single transaction

**Workflow State Machine Integration (AC8-10):**

```java
// In SpeakerOutreachService.recordOutreach()
// SpeakerWorkflowService - state machine for speaker lifecycle
// Location: services/event-management-service/src/main/java/ch/batbern/events/service/SpeakerWorkflowService.java

@Transactional
public OutreachHistoryResponse recordOutreach(
    String eventId,
    UUID speakerId,
    String organizerUsername,
    RecordOutreachRequest request
) {
    // 1. Validate speaker exists and is in valid state
    SpeakerPool speaker = speakerPoolRepository.findById(speakerId)
        .orElseThrow(() -> new NotFoundException("Speaker not found: " + speakerId));

    if (speaker.getStatus() != SpeakerWorkflowState.IDENTIFIED &&
        speaker.getStatus() != SpeakerWorkflowState.OPEN) {
        throw new ValidationException("Speaker not in valid state for contact: " + speaker.getStatus());
    }

    // 2. Create outreach history record
    OutreachHistory outreach = new OutreachHistory();
    outreach.setSpeakerPoolId(speakerId);
    outreach.setContactDate(request.getContactDate());
    outreach.setContactMethod(request.getContactMethod());
    outreach.setNotes(request.getNotes());
    outreach.setOrganizerUsername(organizerUsername);
    outreachHistoryRepository.save(outreach);

    // 3. Transition speaker workflow state (delegates to Story 5.1a workflow service)
    speakerWorkflowService.updateSpeakerWorkflowState(
        speaker.getSessionId(),
        speakerId,
        SpeakerWorkflowState.CONTACTED,
        organizerUsername
    );

    // 4. Publish domain event (DomainEventPublisher from shared-kernel)
    domainEventPublisher.publishEvent(new SpeakerContactedEvent(
        eventId,
        speakerId,
        organizerUsername,
        request.getContactMethod(),
        outreach.getContactDate()
    ));

    return mapToResponse(outreach);
}
```

// web-frontend/src/services/speakerOutreachService.ts
import { apiClient } from '@/services/api/apiClient';
import type { OutreachHistory, RecordOutreachRequest, OutreachHistoryResponse } from '@/types/speaker.types';

class SpeakerOutreachService {
  /**
   * Record speaker outreach (Story 5.3)
   * Follows pattern from EventTypeService and WorkflowService
   */
  async recordOutreach(
    eventId: string,
    speakerId: string,
    request: RecordOutreachRequest
  ): Promise<OutreachHistoryResponse> {
    const response = await apiClient.post(
      `/api/v1/events/${eventId}/speakers/${speakerId}/outreach`,
      request
    );
    return response.data;
  }

  async getOutreachHistory(eventId: string, speakerId: string): Promise<OutreachHistory[]> {
    const response = await apiClient.get(
      `/api/v1/events/${eventId}/speakers/${speakerId}/outreach`
    );
    return response.data;
  }

  async bulkRecordOutreach(
    eventId: string,
    speakerIds: string[],
    request: RecordOutreachRequest
  ): Promise<void> {
    await Promise.all(
      speakerIds.map(speakerId => this.recordOutreach(eventId, speakerId, request))
    );
  }
}

export const speakerOutreachService = new SpeakerOutreachService();
