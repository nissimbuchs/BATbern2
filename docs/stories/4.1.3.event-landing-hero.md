## Story
**As a** public website visitor,
**I want** to see the current event details prominently displayed on the homepage with engaging hero section and logistics,
**so that** I can immediately understand what the event is about and when/where it takes place without navigating to a separate page.

### Backend API Implementation

**Event Management Service - Add Public Endpoint**

**EventController.java** (add new method):
```java
/**
 * Get current published event (PUBLIC ACCESS - no auth required)
 * Used by public website landing page
 */
@GetMapping("/current")
public ResponseEntity<EventDetailDTO> getCurrentEvent(
    @RequestParam(required = false) String include
) {
    Event currentEvent = eventService.getCurrentPublishedEvent();

    if (currentEvent == null) {
        return ResponseEntity.notFound().build();
    }

    // Support resource expansion
    Set<String> includes = include != null
        ? Set.of(include.split(","))
        : Set.of();

    EventDetailDTO dto = eventMapper.toDetailDTO(currentEvent, includes);
    return ResponseEntity.ok(dto);
}
```

**EventService.java** (add method):
```java
public Event getCurrentPublishedEvent() {
    return eventRepository.findFirstByStatusOrderByDateAsc(EventStatus.PUBLISHED)
        .orElse(null);
}
```

**SecurityConfig.java** (update security rules):
```java
@Bean
public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
    http
        .authorizeHttpRequests(auth -> auth
            // PUBLIC ENDPOINTS - No authentication required
            .requestMatchers(GET, "/api/v1/events/current").permitAll()
            .requestMatchers(GET, "/api/v1/events/{id}").permitAll() // Only if PUBLISHED
            .requestMatchers(GET, "/api/v1/events/{id}/sessions").permitAll()

            // AUTHENTICATED ENDPOINTS - Require ORGANIZER role
            .requestMatchers("/api/v1/events").hasRole("ORGANIZER")
            .requestMatchers(POST, "/api/v1/events").hasRole("ORGANIZER")
            .requestMatchers(PUT, "/api/v1/events/**").hasRole("ORGANIZER")
            .requestMatchers(DELETE, "/api/v1/events/**").hasRole("ORGANIZER")

            .anyRequest().authenticated()
        );

    return http.build();
}
```

### Frontend API Integration

**Extend eventApiClient.ts** (add new methods to existing class):
```typescript
// Add to web-frontend/src/services/eventApiClient.ts

/**
 * PUBLIC METHODS - No authentication required
 * These methods work with or without JWT token (optional auth)
 */

/**
 * Get current published event (public access)
 * Used by public website landing page
 */
async getCurrentEvent(options?: { expand?: string[] }): Promise<EventDetail> {
  try {
    const params = new URLSearchParams();
    if (options?.expand && options.expand.length > 0) {
      params.append('include', options.expand.join(','));
    }

    const url = params.toString()
      ? `${EVENT_API_PATH}/current?${params.toString()}`
      : `${EVENT_API_PATH}/current`;

    const response = await apiClient.get<EventDetail>(url);
    return response.data;
  } catch (error) {
    throw this.transformError(error);
  }
}

/**
 * Get sessions for a specific event (public access)
 */
async getEventSessions(eventId: string): Promise<Session[]> {
  try {
    const response = await apiClient.get<Session[]>(
      `${EVENT_API_PATH}/${eventId}/sessions`
    );
    return response.data;
  } catch (error) {
    throw this.transformError(error);
  }
}
```

**Create useCurrentEvent.ts hook:**
```typescript
import { useQuery } from '@tanstack/react-query';
import { eventApiClient } from '@/services/eventApiClient';

export const useCurrentEvent = () => {
  return useQuery({
    queryKey: ['events', 'current'],
    queryFn: () => eventApiClient.getCurrentEvent({
      expand: ['topics', 'venue', 'speakers', 'sessions']
    }),
    staleTime: 5 * 60 * 1000, // 5 minutes
    retry: 2,
  });
};
```

**CountdownTimer.tsx:**
```typescript
import { differenceInDays } from 'date-fns';

interface CountdownTimerProps {
  eventDate: Date;
}

export const CountdownTimer = ({ eventDate }: CountdownTimerProps) => {
  const daysUntil = differenceInDays(eventDate, new Date());

  // Only show if event is within 30 days
  if (daysUntil < 0 || daysUntil > 30) {
    return null;
  }

  return (
    <div className="mb-8 flex items-center gap-3">
      {/* Pulsing blue dot */}
      <div className="relative">
        <div className="h-3 w-3 rounded-full bg-blue-400 animate-pulse" />
        <div className="absolute inset-0 h-3 w-3 rounded-full bg-blue-400 opacity-75 animate-ping" />
      </div>

      {/* Badge */}
      <div className="text-sm font-light text-zinc-300">
        <span className="text-blue-400 font-medium">Next Event</span>
        {' • '}
        {daysUntil === 0 ? 'Today!' : `${daysUntil} day${daysUntil === 1 ? '' : 's'} until event`}
      </div>
    </div>
  );
};
```
