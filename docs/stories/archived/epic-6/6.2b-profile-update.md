## Story

**As a** speaker,
**I want** to view and update my profile (bio, photo, LinkedIn) via the speaker portal,
**so that** my information is accurate for event materials and I don't need to email organizers with updates.

## Architecture Context

### Architecture Patterns

**Referenced from architecture docs (SOURCE OF TRUTH):**

1. **ADR-004** (Factor User Fields from Domain Entities):
   - Speaker stores only `username` - references User entity
   - Bio, firstName, lastName, profilePictureUrl stored in User (Company Service)
   - Expertise, topics, languages, linkedInUrl stored in Speaker (Event Service)
   - Profile update syncs relevant fields to appropriate service

2. **ADR-002** (Generic File Upload):
   - Profile photo upload uses presigned URL pattern
   - S3 path: `users/{username}/profile/photo.{ext}`
   - CloudFront distribution for serving

3. **HTTP Enrichment Pattern**:
   - GET profile combines User + Speaker data
   - PATCH profile routes updates to correct service

### API Specification

**From Architecture Plan (Source of Truth):**

```yaml
GET /api/v1/speaker-portal/profile
  Description: Get speaker profile (combined User + Speaker data)
  Auth: Magic link token

  Query Parameters:
    token: string  # Magic link token

  Response 200 OK:
    # From User entity (Company Service)
    username: string
    email: string
    firstName: string
    lastName: string
    bio?: string
    profilePictureUrl?: string

    # From Speaker entity (Event Service)
    expertiseAreas?: string[]   # Hidden from UI (2026-02-11), API unchanged
    speakingTopics?: string[]   # Hidden from UI (2026-02-11), API unchanged
    linkedInUrl?: string
    languages?: string[]        # Hidden from UI (2026-02-11), API unchanged

    # Computed
    profileCompleteness: number  # 0-100 percentage
    missingFields: string[]      # Fields needed for 100%

  Response 401 Unauthorized:
    error: "TOKEN_EXPIRED" | "TOKEN_NOT_FOUND"

---

PATCH /api/v1/speaker-portal/profile
  Description: Update speaker profile
  Auth: Magic link token

  Request Body:
    token: string

    # User fields (synced to Company Service)
    firstName?: string
    lastName?: string
    bio?: string           # Max 500 chars

    # Speaker fields (stored in Event Service)
    expertiseAreas?: string[]   # Max 10 items - hidden from UI (2026-02-11), API still accepts
    speakingTopics?: string[]   # Max 10 items - hidden from UI (2026-02-11), API still accepts
    linkedInUrl?: string        # Valid URL format
    languages?: string[]        # ISO 639-1 codes - hidden from UI (2026-02-11), API still accepts

  Response 200 OK:
    # Updated profile (same schema as GET)
    username: string
    firstName: string
    # ... all fields

  Response 400 Bad Request:
    error: "VALIDATION_ERROR"
    fields:
      - field: "bio"
        message: "Bio must not exceed 500 characters"

  Response 401 Unauthorized:
    error: "TOKEN_EXPIRED" | "TOKEN_NOT_FOUND"

---

POST /api/v1/speaker-portal/profile/photo/presigned-url
  Description: Get presigned URL for profile photo upload
  Auth: Magic link token

  Request Body:
    token: string
    fileName: string       # Original filename
    contentType: string    # image/jpeg, image/png, image/webp

  Response 200 OK:
    uploadUrl: string      # S3 presigned PUT URL
    uploadId: string       # For confirmation
    expiresIn: number      # Seconds until expiry (900)
    maxSizeBytes: number   # 5MB limit

  Response 400 Bad Request:
    error: "INVALID_CONTENT_TYPE"
    allowedTypes: ["image/jpeg", "image/png", "image/webp"]

---

POST /api/v1/speaker-portal/profile/photo/confirm
  Description: Confirm photo upload and update profile
  Auth: Magic link token

  Request Body:
    token: string
    uploadId: string

  Response 200 OK:
    profilePictureUrl: string  # CloudFront URL

  Response 400 Bad Request:
    error: "UPLOAD_NOT_FOUND" | "UPLOAD_INCOMPLETE"
```

## Dev Notes - Implementation Guide

### Profile Service with HTTP Enrichment

```java
@Service
@RequiredArgsConstructor
public class SpeakerProfileService {

    private final SpeakerRepository speakerRepository;
    private final MagicLinkService magicLinkService;
    private final UserServiceClient userServiceClient;
    private final FileUploadService fileUploadService;

    public SpeakerProfileDto getProfile(String token) {
        // 1. Validate token
        TokenValidationResult tokenResult = magicLinkService.validateToken(token);
        if (!tokenResult.valid()) {
            throw new InvalidTokenException(tokenResult.error());
        }

        // 2. Get Speaker entity
        Speaker speaker = speakerRepository.findByUsername(tokenResult.username())
            .orElseThrow(() -> new SpeakerNotFoundException(tokenResult.username()));

        // 3. Enrich with User data from Company Service
        UserDto user = userServiceClient.getUser(tokenResult.username());

        // 4. Calculate completeness
        int completeness = calculateCompleteness(user, speaker);
        List<String> missingFields = getMissingFields(user, speaker);

        // 5. Build combined response
        return SpeakerProfileDto.builder()
            // User fields
            .username(user.username())
            .email(user.email())
            .firstName(user.firstName())
            .lastName(user.lastName())
            .bio(user.bio())
            .profilePictureUrl(user.profilePictureUrl())
            // Speaker fields
            .expertiseAreas(speaker.getExpertiseAreas())
            .speakingTopics(speaker.getSpeakingTopics())
            .linkedInUrl(speaker.getLinkedInUrl())
            .languages(speaker.getLanguages())
            // Computed
            .profileCompleteness(completeness)
            .missingFields(missingFields)
            .build();
    }

    @Transactional
    public SpeakerProfileDto updateProfile(String token, ProfileUpdateRequest request) {
        TokenValidationResult tokenResult = magicLinkService.validateToken(token);
        if (!tokenResult.valid()) {
            throw new InvalidTokenException(tokenResult.error());
        }

        // Update User fields in Company Service
        if (hasUserFields(request)) {
            userServiceClient.updateUser(tokenResult.username(), UserUpdateDto.builder()
                .firstName(request.firstName())
                .lastName(request.lastName())
                .bio(request.bio())
                .build());
        }

        // Update Speaker fields locally
        if (hasSpeakerFields(request)) {
            Speaker speaker = speakerRepository.findByUsername(tokenResult.username())
                .orElseThrow(() -> new SpeakerNotFoundException(tokenResult.username()));

            if (request.expertiseAreas() != null) {
                validateMaxItems(request.expertiseAreas(), 10, "expertiseAreas");
                speaker.setExpertiseAreas(request.expertiseAreas());
            }
            if (request.speakingTopics() != null) {
                validateMaxItems(request.speakingTopics(), 10, "speakingTopics");
                speaker.setSpeakingTopics(request.speakingTopics());
            }
            if (request.linkedInUrl() != null) {
                validateLinkedInUrl(request.linkedInUrl());
                speaker.setLinkedInUrl(request.linkedInUrl());
            }
            if (request.languages() != null) {
                speaker.setLanguages(request.languages());
            }

            speakerRepository.save(speaker);
        }

        // Return updated profile
        return getProfile(token);
    }

    private int calculateCompleteness(UserDto user, Speaker speaker) {
        int score = 0;
        if (user.firstName() != null && !user.firstName().isBlank()) score += 15;
        if (user.lastName() != null && !user.lastName().isBlank()) score += 15;
        if (user.bio() != null && !user.bio().isBlank()) score += 20;
        if (user.profilePictureUrl() != null) score += 20;
        if (speaker.getExpertiseAreas() != null && !speaker.getExpertiseAreas().isEmpty()) score += 15;
        if (speaker.getLanguages() != null && !speaker.getLanguages().isEmpty()) score += 15;
        return score;
    }
}
```

```typescript
// ProfileForm.tsx
interface ProfileFormProps {
  profile: SpeakerProfile;
  onSave: (updates: ProfileUpdate) => Promise<void>;
  isSaving: boolean;
}

export function ProfileForm({ profile, onSave, isSaving }: ProfileFormProps) {
  const [formData, setFormData] = useState<ProfileUpdate>({
    firstName: profile.firstName,
    lastName: profile.lastName,
    bio: profile.bio || '',
    expertiseAreas: profile.expertiseAreas || [],
    speakingTopics: profile.speakingTopics || [],
    linkedInUrl: profile.linkedInUrl || '',
    languages: profile.languages || [],

  const [isDirty, setIsDirty] = useState(false);

  // Warn on navigation with unsaved changes
  useBeforeUnload(isDirty);

  const handleSubmit = async (e: FormEvent) => {
    e.preventDefault();
    await onSave(formData);
    setIsDirty(false);
  };

    <form onSubmit={handleSubmit} className="space-y-6">
      {/* Basic Info */}

          label="Last Name"
```
