### Database Schema (SQL)

```sql
-- Events table (already exists from Epic 2)
-- See: docs/architecture/03-data-architecture.md, Section 3.2

-- GIN indexes for full-text search (CRITICAL for Story 4.3)
CREATE INDEX IF NOT EXISTS idx_events_search
  ON events USING GIN (to_tsvector('english', title || ' ' || COALESCE(description, '')));

CREATE INDEX IF NOT EXISTS idx_sessions_search
  ON sessions USING GIN (to_tsvector('english', title || ' ' || COALESCE(description, '')));

-- Indexes for archive filtering
CREATE INDEX IF NOT EXISTS idx_events_date
  ON events (date DESC);

CREATE INDEX IF NOT EXISTS idx_events_workflow_state
  ON events (workflow_state);

-- Topic taxonomy (if not already created)
CREATE TABLE IF NOT EXISTS topics (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  code VARCHAR(50) NOT NULL UNIQUE,
  description TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Event-Topic junction table
CREATE TABLE IF NOT EXISTS event_topics (
  event_id UUID REFERENCES events(id) ON DELETE CASCADE,
  topic_id UUID REFERENCES topics(id) ON DELETE CASCADE,
  PRIMARY KEY (event_id, topic_id)
);
```
