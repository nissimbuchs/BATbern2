# Story 4.1.4: Homepage Event Content Sections (Speakers, Sessions, Venue, Social)

## Status
Approved

## Story
**As a** public website visitor,
**I want** to see detailed event content including speakers, sessions, venue map, and social sharing on the homepage,
**so that** I can learn about the program and decide which sessions to attend.

**ARCHITECTURAL NOTE**: All event content sections are integrated into the homepage for a comprehensive single-page event experience. There is no separate `/current-event` page.

## Domain Context

### Primary Domain
Attendee Experience Domain (Public event content discovery)

### Involved Services
- web-frontend (React 19 + TypeScript)
- Event Management Service (via API Gateway)
- API Gateway: Event data already includes speakers, sessions from Story 4.1.3

### Cross-Domain Dependencies
- Speaker Coordination Service (speaker data)
- Session data from Event Management Service

### Related Functional Requirements
- Epic 4, Story 4.1
- AC 5-9: Speaker Grid, Session Cards, Venue Map, Social Sharing

## Acceptance Criteria

1. **API Integration**: Fetch speakers and sessions via `?include=speakers,sessions` (already fetched in 4.1.3)
2. SpeakerGrid displays speaker photos, names, companies, session titles
3. Speaker cards show session abstract on hover
4. SessionCards component with grid/list toggle
5. Session filtering by time/topic
6. Venue map embedded (Google Maps) with event.venue.coordinates
7. Social sharing buttons (LinkedIn, Twitter, Email) with Open Graph metadata
8. EventProgram vertical timeline showing session schedule
9. All data from real API endpoints
10. Responsive design for all components

**SessionCards.tsx** (simplified):
```typescript
import { useState } from 'react';
import { Button } from '@/components/public/ui/button';
import { Card } from '@/components/public/ui/card';
import { Badge } from '@/components/public/ui/badge';
import { LayoutGrid, List, Clock, MapPin, Users } from 'lucide-react';
import type { Session, Topic } from '@/types/event.types';

interface SessionCardsProps {
  sessions: Session[];
  topics: Topic[];
}

export const SessionCards = ({ sessions, topics }: SessionCardsProps) => {
  const [viewMode, setViewMode] = useState<'grid' | 'list'>('grid');
  const [selectedTopics, setSelectedTopics] = useState<string[]>([]);

  const filteredSessions = sessions.filter((session) =>
    selectedTopics.length === 0 ||
    session.topicIds.some((tid) => selectedTopics.includes(tid))
  );

  return (
    <div>
      <div className="flex items-center justify-between mb-6">
        <h2 className="text-3xl font-light">Sessions</h2>
        <div className="flex gap-2">
          <Button
            variant={viewMode === 'grid' ? 'default' : 'outline'}
            size="icon"
            onClick={() => setViewMode('grid')}
          >
            <LayoutGrid className="h-4 w-4" />
          </Button>
          <Button
            variant={viewMode === 'list' ? 'default' : 'outline'}
            size="icon"
            onClick={() => setViewMode('list')}
          >
            <List className="h-4 w-4" />
          </Button>
        </div>
      </div>

      {/* Topic filters */}
      <div className="flex flex-wrap gap-2 mb-6">
        {topics.map((topic) => (
          <Badge
            key={topic.id}
            className={`cursor-pointer ${selectedTopics.includes(topic.id) ? 'bg-blue-400 text-zinc-950' : 'bg-zinc-800'}`}
            onClick={() => setSelectedTopics((prev) =>
              prev.includes(topic.id)
                ? prev.filter((id) => id !== topic.id)
                : [...prev, topic.id]
            )}
          >
            {topic.name}
          </Badge>
        ))}
      </div>

      {/* Sessions */}
      <div className={viewMode === 'grid' ? 'grid grid-cols-1 md:grid-cols-2 gap-4' : 'space-y-4'}>
        {filteredSessions.map((session) => (
          <Card key={session.id} className="p-4">
            <h3 className="font-light text-lg">{session.title}</h3>
            <p className="text-sm text-zinc-400 mt-2">{session.abstract}</p>
            <div className="mt-4 flex flex-wrap gap-4 text-sm text-zinc-400">
              <span className="flex items-center gap-1">
                <Clock className="h-4 w-4" />
                {session.startTime} - {session.endTime}
              </span>
              <span className="flex items-center gap-1">
                <MapPin className="h-4 w-4" />
                {session.room}
              </span>
              <span className="flex items-center gap-1">
                <Users className="h-4 w-4" />
                {session.registrationCount}/{session.capacity}
              </span>
            </div>
            <p className="mt-2 text-sm text-zinc-300">Speaker: {session.speakerName}</p>
          </Card>
        ))}
      </div>
    </div>
  );
};
```

**VenueMap.tsx:**
```typescript
interface VenueMapProps {
  venue: {
    name: string;
    address: string;
    city: string;
    coordinates: { lat: number; lng: number };
  };
}

export const VenueMap = ({ venue }: VenueMapProps) => {
  const mapUrl = `https://www.google.com/maps/embed/v1/place?key=${import.meta.env.VITE_GOOGLE_MAPS_API_KEY}&q=${venue.coordinates.lat},${venue.coordinates.lng}`;
  const directionsUrl = `https://www.google.com/maps/dir/?api=1&destination=${venue.coordinates.lat},${venue.coordinates.lng}`;

  return (
    <div>
      <h2 className="text-3xl font-light mb-6">Venue</h2>
      <div className="rounded-lg overflow-hidden">
        <iframe
          src={mapUrl}
          width="100%"
          height="400"
          style={{ border: 0 }}
          allowFullScreen
          loading="lazy"
          referrerPolicy="no-referrer-when-downgrade"
        />
      </div>
      <div className="mt-4">
        <p className="text-lg font-light">{venue.name}</p>
        <p className="text-zinc-400">{venue.address}, {venue.city}</p>
        <Button asChild variant="outline" className="mt-4">
          <a href={directionsUrl} target="_blank" rel="noopener noreferrer">
            Get Directions
          </a>
        </Button>
      </div>
    </div>
  );
};
```

**SocialSharing.tsx:**
```typescript
import { Button } from '@/components/public/ui/button';
import { Linkedin, Twitter, Mail } from 'lucide-react';

interface SocialSharingProps {
  eventTitle: string;
  eventUrl: string;
}

export const SocialSharing = ({ eventTitle, eventUrl }: SocialSharingProps) => {
  const shareText = `Check out ${eventTitle} at BATbern!`;

  const links = {
    linkedin: `https://www.linkedin.com/sharing/share-offsite/?url=${encodeURIComponent(eventUrl)}`,
    twitter: `https://twitter.com/intent/tweet?text=${encodeURIComponent(shareText)}&url=${encodeURIComponent(eventUrl)}`,
    email: `mailto:?subject=${encodeURIComponent(eventTitle)}&body=${encodeURIComponent(`${shareText} ${eventUrl}`)}`,
  };

  return (
    <div>
      <h2 className="text-3xl font-light mb-6">Share Event</h2>
      <div className="flex gap-4">
        <Button asChild variant="outline">
          <a href={links.linkedin} target="_blank" rel="noopener noreferrer">
            <Linkedin className="h-4 w-4 mr-2" />
            LinkedIn
          </a>
        </Button>
        <Button asChild variant="outline">
          <a href={links.twitter} target="_blank" rel="noopener noreferrer">
            <Twitter className="h-4 w-4 mr-2" />
            Twitter
          </a>
        </Button>
        <Button asChild variant="outline">
          <a href={links.email}>
            <Mail className="h-4 w-4 mr-2" />
            Email
          </a>
        </Button>
      </div>
    </div>
  );
};
```
