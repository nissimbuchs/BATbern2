### Unit Testing Example

**EventHero.test.tsx:**
```typescript
import { render, screen } from '@testing-library/react';
import { BrowserRouter } from 'react-router-dom';
import { EventHero } from '@/components/public/Event/EventHero';

const mockEvent = {
  id: 'BATbern025',
  title: 'Cloud-Native Architecture',
  description: 'An evening of architecture discussions',
};

describe('EventHero', () => {
  it('should render event title', () => {
    render(
      <BrowserRouter>
        <EventHero event={mockEvent} />
      </BrowserRouter>
    );
    expect(screen.getByText('Cloud-Native Architecture')).toBeInTheDocument();
  });

  it('should render register button', () => {
    render(
      <BrowserRouter>
        <EventHero event={mockEvent} />
      </BrowserRouter>
    );
    expect(screen.getByText('Register Now')).toBeInTheDocument();
  });

  it('should link to registration page', () => {
    render(
      <BrowserRouter>
        <EventHero event={mockEvent} />
      </BrowserRouter>
    );
    const link = screen.getByText('Register Now').closest('a');
    expect(link).toHaveAttribute('href', '/register/BATbern025');
  });
});
```

### E2E Testing Example

**e2e/registration-flow.spec.ts:**
```typescript
import { test, expect } from '@playwright/test';

test.describe('Event Registration Flow', () => {
  test('should complete full registration', async ({ page }) => {
    // 1. Browse event
    await page.goto('http://localhost:3000/current-event');
    await expect(page.getByText('Cloud-Native Architecture')).toBeVisible();

    // 2. Click register
    await page.click('text=Register Now');
    await expect(page).toHaveURL(/\/register\//);

    // 3. Fill personal info (Step 1)
    await page.fill('input[name="firstName"]', 'John');
    await page.fill('input[name="lastName"]', 'Doe');
    await page.fill('input[name="email"]', 'john@example.com');
    await page.fill('input[name="company"]', 'Acme Corp');
    await page.fill('input[name="role"]', 'Architect');
    await page.click('text=Next');

    // 4. Select sessions (Step 2)
    await page.click('[data-testid="session-checkbox"]:first-child');
    await page.click('text=Next');

    // 5. Review and accept terms (Step 3)
    await page.click('[data-testid="terms-checkbox"]');
    await page.click('text=Complete Registration');

    // 6. Verify confirmation page
    await expect(page).toHaveURL(/\/registration-confirmation\//);
    await expect(page.getByText('Registration Confirmed!')).toBeVisible();

    // 7. Verify QR code exists
    await expect(page.locator('img[alt*="QR Code"]')).toBeVisible();
  });

  test('should show validation errors', async ({ page }) => {
    await page.goto('http://localhost:3000/register/BATbern025');
    await page.click('text=Next'); // Try to proceed without filling form
    await expect(page.getByText('First name is required')).toBeVisible();
  });
});
```

### SEO Meta Tags

**MetaTags.tsx:**
```typescript
import { Helmet } from 'react-helmet-async';

interface MetaTagsProps {
  title: string;
  description: string;
  image?: string;
  url?: string;
  type?: 'website' | 'article';
}

export const MetaTags = ({ title, description, image, url, type = 'website' }: MetaTagsProps) => {
  const fullTitle = `${title} | BATbern`;
  const defaultImage = 'https://batbern.ch/og-image.jpg';
  const defaultUrl = 'https://batbern.ch';

  return (
    <Helmet>
      {/* Basic Meta */}
      <title>{fullTitle}</title>
      <meta name="description" content={description} />
      <link rel="canonical" href={url || defaultUrl} />

      {/* Open Graph */}
      <meta property="og:type" content={type} />
      <meta property="og:title" content={fullTitle} />
      <meta property="og:description" content={description} />
      <meta property="og:image" content={image || defaultImage} />
      <meta property="og:url" content={url || defaultUrl} />
      <meta property="og:site_name" content="BATbern" />

      {/* Twitter Card */}
      <meta name="twitter:card" content="summary_large_image" />
      <meta name="twitter:title" content={fullTitle} />
      <meta name="twitter:description" content={description} />
      <meta name="twitter:image" content={image || defaultImage} />
    </Helmet>
  );
};
```

**Usage in CurrentEventPage:**

### JSON-LD Structured Data

**EventSchema.tsx:**
```typescript
import { Helmet } from 'react-helmet-async';
import { Event } from '@/services/eventApi';

interface EventSchemaProps {
  event: Event;
}

export const EventSchema = ({ event }: EventSchemaProps) => {
  const schema = {
    '@context': 'https://schema.org',
    '@type': 'Event',
    name: event.title,
    description: event.description,
    startDate: `${event.startDate}T${event.startTime}`,
    endDate: `${event.endDate}T${event.endTime}`,
    eventStatus: 'https://schema.org/EventScheduled',
    eventAttendanceMode: 'https://schema.org/OfflineEventAttendanceMode',
    location: event.venue ? {
      '@type': 'Place',
      name: event.venue.name,
      address: {
        '@type': 'PostalAddress',
        streetAddress: event.venue.address,
        addressLocality: event.venue.city,
        addressCountry: 'CH',
      },
      geo: {
        '@type': 'GeoCoordinates',
        latitude: event.venue.coordinates.lat,
        longitude: event.venue.coordinates.lng,
      },
    } : undefined,
    offers: {
      '@type': 'Offer',
      price: '0',
      priceCurrency: 'CHF',
      availability: 'https://schema.org/InStock',
      url: `https://batbern.ch/register/${event.id}`,
    },
    organizer: {
      '@type': 'Organization',
      name: 'Berner Architekten Treffen',
      url: 'https://batbern.ch',
    },
  };

  return (
    <Helmet>
      <script type="application/ld+json">
        {JSON.stringify(schema)}
      </script>
    </Helmet>
  );
};
```
